\begin{frame}{The story of this talk}
  \begin{enumerate}
    \item<1-> \textbf{Problem:} real devices leak; measuring power or timing
      can recover secret keys. \textbf{Standard fix:} masking --- compute on
      randomized shares, never on secrets.
    \item<2-> \textbf{The question:} was the masking done right? ``Right''
      means no $d$ observed wires can tell two secrets apart.
    \item<3-> \textbf{It is provably hard:} as hard as counting solutions over
      all input combinations, so no tool can simply compute the answer.
    \item<4-> \textbf{Our route:} prove security by re-arranging the
      randomness until the secrets visibly cancel. \toolname{} does this in
      Lean~4, with a rewrite engine, a combinatorial search, and reusable
      certificates.
    \item<5-> \textbf{Result:} soundness proofs, gadgets verified that a
      comparable tool false-flags, and competitive performance.
  \end{enumerate}
\end{frame}

\section{Motivation \& Model}
% =====================================================================

\begin{frame}{Side-channel attacks}
  A cipher can be secure on paper and insecure on the bench.
  \begin{itemize}
    \item A real device leaks through power draw, timing, electromagnetic
      emanation; all of which vary with the \emph{data} it handles.
    \item Differential power analysis \parencite{kocher1999differential} correlates many power
      traces against a guess of an intermediate value, and may recover the key.
    \item In practice: DPA recovered keys from deployed smartcards; timing
      channels underlie Spectre and Meltdown.
  \end{itemize}
  \medskip
  \begin{alertblock}{}
    So, the insecurity lies not in the mathematics behind the cipher, but in the way
    the hardware computes it.
  \end{alertblock}
\end{frame}

\begin{frame}{The setting}
  We study \textbf{arithmetic circuits} over the two-element field $\F$, built
  from two-input $\oplus$ (XOR) and $\odot$ (AND) gates.
  \medskip
  \begin{block}{Inputs are partitioned into three classes}
    \begin{itemize}
      \item {\color{hlbad}\textbf{secrets} $\Ss$} --- the key material we protect,
      \item {\color{hlrand}\textbf{randoms} $\Rs$} --- fresh uniform bits (the
        masks), drawn together as a \emph{tape} $\rho \in \{0,1\}^{\Rs}$,
      \item {\color{black!60}\textbf{publics} $\mathcal{P}$} --- known to
        everyone, held fixed.
    \end{itemize}
  \end{block}
  \medskip
  Leakage is \emph{value-based}: a probed wire reveals the logical value it
  carries and nothing more --- no glitches, no transitions.
\end{frame}

\begin{frame}{Masking --- and our running example}
  \emph{Boolean masking} \parencite{chari1999towards}: split each secret into
  shares, $x = x_0 \oplus x_1$ with {\color{hlrand}$x_0$ uniformly random},
  and compute only on shares, never on secrets.
  \medskip
  \begin{columns}[c]
    \begin{column}{0.52\textwidth}
      \centering
      \domand{}\\[1ex]
      {\scriptsize {\color{hlrand}$\blacksquare$}~random \quad
       {\color{hlbad}$\blacksquare$}~secret-carrying \quad
       $\square$~computed}
    \end{column}
    \begin{column}{0.44\textwidth}
      \textbf{DOM-AND} \parencite{gross2016domain}: a masked AND gate, two
      shares per input, one fresh mask $r$.
      \begin{itemize}
        \item outputs re-share $s_0 \oplus s_1 = x \odot y$,
        \item cross terms $p_{01}, p_{10}$ mix shares of \emph{both} secrets,
        \item $r$ re-masks them before the outputs.
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{The probing model}
  \small
  Reasoning about noisy traces directly is unwieldy. Luckily we have an
  abstraction \parencite{10.1007/978-3-540-45146-4_27}:
  \begin{block}{$d$-probing security}
    An adversary picks any $d$ wires $W = \{w_1, \dots, w_d\}$ and is handed
    their \emph{exact} values. The circuit is $d$-probing secure when, for
    every such choice and every two secret assignments $\va, \vb$, the
    observed values are identically distributed over the tape $\rho$:
    \[
      \law{\va}(w_1, \dots, w_d) \;=\; \law{\vb}(w_1, \dots, w_d) .
    \]
  \end{block}
  \vspace{-1ex}
  \begin{columns}[c]
    \begin{column}{0.5\textwidth}
      \centering
      \domand{wc01/.style={wire,probed}}
    \end{column}
    \begin{column}{0.46\textwidth}
      \begin{itemize}
        \item<2-> Security becomes a \emph{combinatorial} property,
          decidable in principle.
        \item<3-> The abstraction is faithful: probing security implies
          noisy-leakage security \parencite{duc2014unifying}.
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{The problem}
  \begin{block}{Exact verification of $d$-probing security}
    Decide, for \emph{every} one of the $\sum_{i \le d} \binom{|\Ws|}{i}$
    observation sets $O$, whether it is \emph{independent of the secrets}:
    \[
      \law{\va}(O) \;=\; \law{\vb}(O)
      \qquad \text{for all } \va, \vb \in \{0,1\}^{\Ss} .
    \]
    \begin{itemize}
      \item $\Secure$ \;\;\; $\Rightarrow$ every observation set accounted for,
      \item $\Insecure$ \; $\Rightarrow$ a specific leaking observation as
        witness.
    \end{itemize}
  \end{block}
  \medskip
  \begin{itemize}
    \item<2-> Two difficulties arise: a \emph{single} probe already asks a question
      about a distribution over all of $\{0,1\}^{\Rs}$,
    \item<2-> and there are combinatorially many probes to settle.
  \end{itemize}
\end{frame}

\section{How Hard Is One Probe?}
% =====================================================================

\begin{frame}{Warm-up: three views of secret-independence}
  Fix a probe $O$ (a tuple $\mathbf{w}$ of wires). Write $\law{\va}(O)$ for its
  distribution over a uniform tape $\rho \in \{0,1\}^{\Rs}$, and
  $N_{\va}(o) = |\{\rho : O(\rho;\va) = o\}|$ for the tape counts.
  \medskip
  \begin{enumerate}
    \item<1-> \textbf{Distributional.} \; $\law{\va}(O) = \law{\vb}(O)$ for all
      secrets $\va, \vb$. \hfill {\small \emph{(the definition)}}
    \item<2-> \textbf{Counting.} \; $N_{\va}(o) = N_{\vb}(o)$ for every
      observation $o$, i.e. two histograms of integers coincide.
      
      {\small \emph{$\to$ this section}}
    \item<3-> \textbf{Coupling.} \; there exists a bijection
      $\Phi$ of the tapes such that $O(\rho;\va) = O(\Phi(\rho);\vb)$.
      
      {\small \emph{$\to$ next section}}
  \end{enumerate}
  % \medskip
  % \onslide<4->{
  % All three are equivalent --- but they are \emph{not} equally usable.
  % The counting view tells us what a verification procedure should not attempt
  % directly; the coupling view tells us what it might be able to acomplish.}
\end{frame}

\begin{frame}{Counting is intractable}
  \begin{block}{Theorem (thesis, Ch.~4)}
    Computing one tape count $N_{\va}(o)$ is $\sharpP$-complete
    \parencite{valiant1979complexity}. And deciding $\law{\va}(O) = \law{\vb}(O)$ is
    $\CeqP$-complete \parencite{wagner1986complexity}.
  \end{block}
  \medskip
  \begin{itemize}
    \item<2-> Hardness: reduction from \textsc{ExactSat} --- one secret bit
      and one probed wire already encode it.
      \hfill {\small(construction @ the appendix slides)}
    \item<2-> So the hardness is intrinsic; no feature of masking causes it.
  \end{itemize}
\end{frame}

\begin{frame}{What the placement tells}
  \begin{itemize}
    \item Deciding never costs more than counting
      ($\CeqP \subseteq \mathrm{P}^{\sharpP}$), and no polynomial exact
      decision exists unless $\mathrm{P} = \mathrm{NP}$.
    \item Both classes sit \emph{above the polynomial hierarchy} (due to Toda).
      A single probe is already there, and full verification universally
      quantifies over exponentially many such instances.
  \end{itemize}
  \medskip
  \begin{exampleblock}{Consequence}
    We do not compute counts. We certify \emph{constructively}: exhibit a
    relabelling of the randomness under which the secrets visibly cancel ---
    a witness without counting, at the price of completeness!
  \end{exampleblock}
\end{frame}

\section{Security Is a Coupling}
% =====================================================================

\begin{frame}{One substitution, on the running example}
  \small
  Probe the masked cross term: $c_{01} = x_0 \odot y_1 \oplus r$.
  \medskip
  \begin{columns}[c]
    \begin{column}{0.5\textwidth}
      \centering
      \domand{wc01/.style={wire,probed}, wr/.style={wire,randw,safe}}
    \end{column}
    \begin{column}{0.46\textwidth}
      Apply the substitution
      $\sigma = \bigl(r \mapsto x_0 \odot y_1 \oplus r\bigr)$:
      \[
        c_{01}\,\sigma
        \;=\; x_0 \odot y_1 \oplus (x_0 \odot y_1 \oplus r)
        \;=\; r .
      \]
      \begin{itemize}
        \item<2-> The probe is now \emph{syntactically secret-free}: it is
          uniform, whatever the secrets are.
        \item<3-> And $\sigma$ induces a \emph{bijection} $\widehat{\sigma}$
          of the tape space, translating one coordinate --- a \emph{shear}.
      \end{itemize}
    \end{column}
  \end{columns}
  \medskip
  \onslide<3->{Key identity: \; $(\mathbf{w}\sigma)(\rho) = \mathbf{w}(\widehat{\sigma}(\rho))$
  --- the syntactic step and the semantic relabelling agree.}
\end{frame}

\begin{frame}{Couplings}
  To prove two distributions equal \emph{without computing them}, link their
  sources (\parencite{barthe2017coupling} for probabilistic programs). Our two
  laws arise from the \emph{same} uniform tape, so we link the tapes.
  \medskip
  \begin{block}{Definition (coupling function)}
    A \emph{coupling function} from $\va$ to $\vb$ for a probe $O$ is a
    bijection $\Phi : \{0,1\}^{\Rs} \to \{0,1\}^{\Rs}$ with
    \[
      O(\rho;\va) \;=\; O\bigl(\Phi(\rho);\vb\bigr)
      \qquad \text{for every tape } \rho .
    \]
  \end{block}
  \begin{itemize}
    \item<1-> Bijectivity preserves the uniform law; the identity says a run
      under $\va$ is reproduced, observation for observation, under $\vb$.
      Hence we have equal laws.
    % \item<3-> The family $\Phi = \{\Phi_{\va,\vb}\}$ over ordered secret pairs
    %   forms a \emph{groupoid}: $\Phi_{\va,\va} = \mathrm{id}$,
    %   $\Phi_{\vb,\va} = \Phi_{\va,\vb}^{-1}$,
    %   $\Phi_{\vb,\mathbf{c}} \circ \Phi_{\va,\vb} = \Phi_{\va,\mathbf{c}}$.
    %   One anchor row generates the rest.
  \end{itemize}
\end{frame}

\begin{frame}{Security \emph{is} a coupling}
  \small
  \begin{block}{Theorem}
    A probe $O$ is secret-independent \emph{iff} for every ordered pair
    $\va, \vb$ there is a coupling function from $\va$ to $\vb$ for $O$.
  \end{block}
  \medskip
  \emph{Proof sketch.}
  \begin{itemize}
    \item ($\Leftarrow$) A bijection merely re-indexes the tape space, so the
      counts behind the two laws agree.
    \item ($\Rightarrow$) Equal laws mean the outcome-fibres
      $\{\rho : O(\rho;\va) = o\}$ come in matching sizes; match them block
      by block and glue. \qed
      
      \hfill {\small(full sketch: backup slides)}
  \end{itemize}
  \medskip
  \onslide<2->{
  \begin{alertblock}{This is an existence statement and not a procedure}
    Matching the fibres needs the counts $N_{\va}(o)$, which are
    $\sharpP$-hard. The theorem guarantees a coupling; it does not find one.
  \end{alertblock}}
\end{frame}

\begin{frame}{From substitutions to couplings}
  \small
  A derivation $\sigma_1, \dots, \sigma_K$ ending secret-free is a
  \emph{recipe}: run under secrets $\va$, it becomes the tape bijection
  $\varphi_{\va} = \widehat{\sigma_1}^{\va} \circ \cdots \circ
  \widehat{\sigma_K}^{\va}$. Applied syntactically, it yields $\overline{O} = O\sigma_1\dots\sigma_K$.
  \begin{center}
  \begin{tikzpicture}[
      exp/.style={font=\small, inner sep=3pt},
      map/.style={-{Stealth[length=2mm]}, thick},
      lbl/.style={font=\footnotesize}]
    \node[exp] (Oa) at (0,0)   {$O(\,\cdot\,;\va)$};
    \node[exp] (Ob) at (7.6,0) {$O(\,\cdot\,;\vb)$};
    \node[exp] (nf) at (3.8,-1.6) {\alert{$\overline{O}$}};
    \node[lbl, right=1mm of nf, text=black!60] {\;(secret-free normal form)};
    \draw[map] (Oa) -- node[lbl, midway, left=2pt]{$\varphi_{\va}$} (nf);
    \draw[map] (Ob) -- node[lbl, midway, right=2pt]{$\varphi_{\vb}$} (nf);
    \draw<2->[map, dashed, hlsafe] (Oa) --
      node[lbl, midway, above]{$\varphi_{\vb} \circ \varphi_{\va}^{-1} = \Phi$}
      (Ob);
  \end{tikzpicture}
  \end{center}
  \begin{itemize}
    \item<2-> Every run is a re-labelled view of the \emph{same} secret-free
      experiment, so any two runs link through it: a coupling for
      \emph{every} pair of secrets, from one derivation.
    \item<3-> On DOM-AND the recipe is the single substitution
      $r \mapsto x_0 \odot y_1 \oplus r$; its context $x_0 \odot y_1$ is what
      makes $\varphi$ depend on the secrets.
    \item<4-> So a derivation is a \textbf{reusable certificate}: the design
    principle of \toolname.
  \end{itemize}
\end{frame}

  \medskip
  \begin{itemize}
    \item<2-> \textbf{I} decides a \emph{single} probe, by construction.
    \item<3-> \textbf{II} sweeps \emph{every} $d$-subset without explicitly enumerating
      them.
    \item<4-> \textbf{III} amplifies one certificate into a whole \emph{safe
      set}, shrinking the space \textbf{II} must sweep.
  \end{itemize}
\end{frame}

\begin{frame}{Component I: the rewrite engine}
  Decides one probe $\mathbf{w}$ by substituting randoms away, one at a time, driving the
  observation toward a secret-free form.
  \medskip
  \begin{itemize}
    \item<1-> \textbf{Tiered.} A cheap single-occurrence rule first ($r$ occurs
      once across $\mathbf{w}$: substitute, done); with a complete substitution loop
      behind it.
    \item<2-> \textbf{Sound.} Every step is a shear, so any $\Secure$ verdict
      carries a genuine coupling (Theorem, Ch.~5).
    \item<3-> \textbf{Incomplete, knowingly.} A probe can be independent for
      reasons no relabelling of this kind captures --- this is the price of the
      hardness result.
  \end{itemize}
  \medskip
  \onslide<4->{
  On the example: $c_{01} \xrightarrow{\;\;r \;\mapsto\; x_0 \odot y_1 \oplus r\;\;} r$
  --- one step, $\Secure$, coupling recorded.}
\end{frame}

\begin{frame}{Component I: why factoring}
  Some masks are buried inside \emph{products}, where substitution cannot
  reach them.
  \medskip
  \begin{itemize}
    \item<1-> Substituting $r \mapsto e \oplus r$ needs $r$ to occur as a
      \emph{summand}; in $u \odot r \oplus u \odot v$ every occurrence of $r$
      is trapped inside a product.
    \item<2-> \textbf{Multiplicative factoring} pulls the common factor out,
      \[
        u \odot r \,\oplus\, u \odot v \;\longmapsto\; u \odot (r \oplus v) ,
      \]
      and the hidden random surfaces as a summand again, ready to substitute.
    \item<3-> Applied lazily between substitution attempts, until the probe
      discharges or no rule applies.
  \end{itemize}
  % \medskip
  % \onslide<4->{\emph{Caveat, stated openly:} factoring itself is not a shear
  % --- it changes the syntax, not the tapes. This has a consequence for how we
  % extend probes, coming up in Component III.}
\end{frame}

\begin{frame}{Component II: the representation}
  Certifying each $d$-subset in turn is infeasible. So the traversal never
  lists probes; it holds compact descriptions of whole families, in the manner
  of maskVerif \parencite{10.1007/978-3-030-29959-0_15}.
  \medskip
  \begin{block}{Factors and worklists}
    A \emph{factor} $(c, W)$ stands for ``choose $c$ wires from the set $W$''.
    A \emph{worklist} $\langle (c_1, W_1), \dots, (c_m, W_m) \rangle$, the
    $W_j$ disjoint, denotes all probes drawing $c_j$ wires from each $W_j$:
    it holds $\prod_j \binom{|W_j|}{c_j}$ probes in a few words.
  \end{block}
  \medskip
  On DOM-AND \emph{(order $2$, for exposition)}: the probe targets are the
  eight computed wires
  $\Ws = \{u_0, u_1, p_{01}, p_{10}, c_{01}, c_{10}, s_0, s_1\}$, and the
  whole verification task is the single factor
  \[
    \bigl\langle (2, \Ws) \bigr\rangle ,
    \qquad \text{denoting all } \tbinom{8}{2} = 28 \text{ pairs.}
  \]
  % \onslide<2->{No list of $28$ anywhere --- and the description splits as
  % nicely as it reads.}
\end{frame}

\begin{frame}{Component II: one recursion step, on our example}
  \small
  \begin{columns}[c]
    \begin{column}{0.44\textwidth}
      \centering
      \domand{wu0/.style={wire,safe}, wc01/.style={wire,safe},
              ws0/.style={wire,safe}}\\[0.5ex]
      {\scriptsize the illustrated safe set, in green}
\end{column}
    \begin{column}{0.52\textwidth}
      \begin{enumerate}
        \item<1-> \textbf{Prune?} Discharge the union of all $8$ wires at
          once. It fails here (the gadget leaks at order $2$), so we descend.
        \item<2-> \textbf{Certify a representative probe}, say
          $\{u_0, c_{01}\}$, and grow it (Component III) into a safe set,
          here $\{u_0, c_{01}, s_0\}$.
        \item<3-> \textbf{Split} along safe / unsafe:
          $28 = \tbinom{3}{2} + 3 \cdot 5 + \tbinom{5}{2} = 3 + 15 + 10$
          (all-safe, mixed, all-unsafe). \emph{Skip} the all-safe cell and
          recurse on the $25$ that remain.
        % \item<4-> Eventually $\{c_{01}, c_{10}\}$ reaches Component I:
        %   $\Insecure$, with the witness we know.
      \end{enumerate}
    \end{column}
  \end{columns}
  \smallskip
  \onslide<4->{\textbf{Theorem (exhaustiveness, Ch.~5).} Read set-wise,
  Vandermonde's identity says the cells \emph{partition} the probes: every
  $d$-subset is visited exactly once.}
\end{frame}

\begin{frame}{Component III: growing safe sets}
  A discharged probe is a coupling --- but can the \emph{same} coupling cover more
  wires?
  \medskip
  \begin{columns}[c]
    \begin{column}{0.5\textwidth}
      \centering
      \domand{wc01/.style={wire,safe}, wu0/.style={wire,safe},
              wu1/.style={wire,safe}, wp01/.style={wire,safe}}
    \end{column}
    \begin{column}{0.46\textwidth}
      \begin{itemize}
        \item<1-> A wire $u$ joins the safe set of coupling $\varphi$ when the
          composite $u \circ \varphi$ is secret-free.
        \item<2-> Exact test via \emph{ANF}: a function depends on a variable
          iff it occurs in its normal form.
        \item<3-> One discharge settles the whole safe set; Component II
          needs less traversal.
      \end{itemize}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Component III: three mechanisms}
  \begin{itemize}
    \item<1-> \textbf{Closure} --- build-time, entropy-style deduction
      ($H(u \mid \vw) = 0$). Local deduction on a scattered pool: cheap but
      weak.
    \item<2-> \textbf{Replay} --- re-run the derivation on the candidate wire
      \parencite{barthe2015verified}. An imperfect fit for us because we \emph{factor},
      and factoring is not a replayable step.
    \item<3-> \textbf{Coupling} --- apply the recorded tape bijection and test
      the ANF endpoint. Exact, and independent of how the derivation was
      found.
  \end{itemize}
  \medskip
  \begin{block}{Proposition (nesting, Ch.~5)}
    $\text{closure} \subseteq \text{replay} \subseteq \text{coupling}$ ---
    every wire the cheaper mechanisms admit, the coupling admits too.
    \toolname{} adopts the coupling approach.
  \end{block}
\end{frame}

\begin{frame}{Component III: the test}
  \small
  Given the coupling $\varphi$ and a candidate wire $u$ from the pool:
  is $u \circ \varphi$ secret-free?
  \begin{block}<1->{Phase 1 --- fast reject (probabilistic, may only \emph{reject})}
    Sample tapes; evaluate $u \circ \varphi$ twice per tape, secrets drawn at
    random vs.\ all zero. A disagreement proves $u \circ \varphi$ moves with
    the secrets: drop $u$. Agreement proves nothing.
  \end{block}
  \begin{block}<2->{Phases 2--3 --- exact ($\F$ cancellation, then ANF)}
    A function depends on a variable iff the variable occurs in its unique
    multilinear normal form. So admit $u$ exactly when no secret survives in
    the ANF of $u \circ \varphi$ (Lemma, Ch.~5).
  \end{block}
  \medskip
  \onslide<3->{\textbf{Why blinded wires are safe to add.} For admitted $u$,
  $u(\varphi_{\va}(\rho); \va) = \overline{u}(\rho)$ depends on $\rho$ alone;
  as $\varphi_{\va}$ preserves the uniform law, the \emph{joint} law of the
  safe set is the same under every secret (Theorem, Ch.~5).}
\end{frame}

\section{Implementation \& Evaluation}
% =====================================================================

\begin{frame}{Implementation details}
  \toolname{} is implemented in \textbf{Lean 4}.
  \medskip
  \begin{itemize}
    \item \textbf{Hash-consed DAG} expression representation with
      canonicalization --- structural sharing makes equality checks and
      substitutions cheap.
    \item \textbf{Multiplicative factoring} as a normal-form pass over the
      DAG, applied lazily.
    \item \textbf{Bit-sliced ANF endpoint test} for the extension:
      word-parallel evaluation decides secret-freeness of $u \circ \varphi$
      exactly.
  \end{itemize}
  \medskip
  \onslide<2->{Design and proofs are implementation-agnostic; these choices
  simply make algorithms fast.}
\end{frame}

  \begin{frame}{Evaluation: outcomes}
  \small
  Forty configurations at orders 1--4: ISW multiplications, DOM and TI ANDs,
  refresh gadgets, Keccak $\chi$ (DOM and TI sharings), and the TI S-boxes and
  quadratic sharings of Bilgin et al. Every verdict matches the ground truth,
  and every $\Insecure$ comes with a concrete witness observation.
  \medskip
  \begin{itemize}
    \item<1-> \textbf{DOM-AND} (2 shares): $\Secure$ at order 1, the witness
      pair $\{c_{01}, c_{10}\}$ at order 2 --- our running example.
    \item<1-> \textbf{Four-share $x \oplus yz$ TI} \parencite{bilgin2015threshold}:
      certified secure at orders 1 and 2, insecure at order 3 --- the
      \emph{expected} behaviour of a first-order threshold implementation.
    \item<2-> \textbf{The factoring family} ($d{+}1$ quadratic S-boxes,
      $Q^3\!/Q^4$): randomness-free sharings whose certification must factor
      first. A factoring-free, maskVerif-style engine false-reports
      $\Insecure$ on five; \toolname{} verifies every one.
  \end{itemize}
  \medskip
  \onslide<2->{(Full extension-comparison table @ the appendix slides.)}
\end{frame}

\begin{frame}{Evaluation: is the extension worth it?}
  \small
  \begin{columns}[c]

\section{Wrap-up}
% =====================================================================

\begin{frame}{Limitations and future work}
  \begin{itemize}
    \item<1-> \textbf{Model.} Our model is value-based leakage only: glitches and
      transitions are out of scope; so is composition. So we show single-circuit, exact
      probing security.
    \item<2-> \textbf{Larger fields.} The core --- shears, couplings, ANF ---
      is already field-agnostic; lifting the implementation beyond $\F$ is
      engineering plus a normal-form choice. (Also, the power of Schwartz--Zippel
      can come into picture here.)
    \item<3-> \textbf{Composing couplings.} for unions of safe sets using
      couplings as a guide.
    \item<4-> \textbf{More.} smarter factoring heuristics; exploiting circuit
      symmetries to quotient the subset space.
  \end{itemize}
\end{frame}

\begin{frame}{Contributions}
  \begin{enumerate}
    \item<1-> \textbf{Placement.} We show that: counting is
      $\sharpP$-complete, deciding is $\CeqP$-complete --- above the
      polynomial hierarchy, ruling out the counting route.
    \item<2-> \textbf{A rigorous coupling account of probing security.}
      Security \emph{is} the existence of a family of coupling functions, the
      family is a groupoid, and substitutions construct it.
    \item<3-> \textbf{\toolname.} Made up of a sound rewrite engine (with factoring), an
      exhaustive subset-space traversal,
      and a probe-set extension with a proven mechanism hierarchy
      $\text{closure} \subseteq \text{replay} \subseteq \text{coupling}$.
    \item<4-> \textbf{Evaluation.} We present exact verdicts, with witnesses, across a
      suite of masked gadgets, and the coupling extension wins in coverage.
  \end{enumerate}
\end{frame}

\begin{frame}{Backup: $\CeqP$ membership, sketch}
  Encode disagreement of the two laws as a single gap function:
  \[
    f(C, O, \va, \vb) \;=\; \sum_{o} \bigl(N_{\va}(o) - N_{\vb}(o)\bigr)^2 .
  \]
  Each $N$ is a $\sharpP$ function; $\mathrm{GapP}$ is closed under subtraction
  and multiplication \parencite{fenner1994gap}; the sum has $2^k \le 2^d$ terms,
  constant for fixed $d$. A sum of squares vanishes iff every term does, so
  \[
    \law{\va}(O) = \law{\vb}(O) \iff f = 0 ,
  \]
  and the problem is the vanishing set of a gap function, which is exactly $\CeqP$.
\end{frame}

\begin{frame}{Backup: the ExactSat reduction}
  Reduce from the equality form of \textsc{ExactSat}: given formulas
  $\varphi_0, \varphi_1$ over variables $x_1, \dots, x_m$, decide
  $\#\varphi_0 = \#\varphi_1$ ($\CeqP$-complete,
  \parencite{wagner1986complexity}).
  \medskip

  Build a circuit with one secret bit $s$, the $x_i$ as randoms, and the
  single wire
  \[
    w \;=\; \varphi_0 \,\oplus\, \bigl(s \odot (\varphi_0 \oplus \varphi_1)\bigr),
  \]
  realising the formulas over $\oplus$ and $\odot$. Then $w$ reads $\varphi_0$
  when $s = 0$ and $\varphi_1$ when $s = 1$, so the one-bit probe $\{w\}$ is
  secret-independent \emph{iff} $\#\varphi_0 = \#\varphi_1$.
  \medskip

  The construction is polynomial, and one secret bit with one probed wire
  suffices: no feature of masking is what makes the problem hard.
\end{frame}

\begin{frame}{Backup: security is a coupling, full sketch}
  \small
  ($\Leftarrow$) Reindex by $\rho' = \varphi(\rho)$:
  \[
    N_{\va}(o) = \bigl|\{\rho : O(\rho;\va) = o\}\bigr|
    = \bigl|\{\rho : O(\varphi(\rho);\vb) = o\}\bigr|
    = N_{\vb}(o).
  \]
  \medskip

  ($\Rightarrow$) Sort the tapes by outcome:
  $F_{\va}(o) = \{\rho : O(\rho;\va) = o\}$. As $o$ varies these
  \emph{fibres} partition $\{0,1\}^{\Rs}$, and equal laws mean matching
  fibres have equal size, $|F_{\va}(o)| = |F_{\vb}(o)|$. So pick, for each
  $o$, any bijection $F_{\va}(o) \to F_{\vb}(o)$; their union is a bijection
  $\varphi$ of the whole tape space, and it sends every $\rho$ to a tape with
  the \emph{same} observation, giving us a coupling function. \qed
\end{frame}

\begin{frame}{Backup: replay vs.\ factoring}
  The derivations of \textcite{barthe2015verified} replay two rule kinds: \textsc{Opt}
  (substitution) and \textsc{Conv} (ANF normalisation) --- for them, ANF is a
  \emph{replayed rule}.
  \medskip

  Our derivations interleave substitutions with \emph{multiplicative
  factoring}, and a factoring step has no tape-level counterpart to replay.
  Replaying the substitution steps alone on a new wire can spuriously fail,
  producing false-$\Insecure$ verdicts (observed in the evaluation, the
  $\dagger$ rows).
  \medskip

  Hence the coupling mechanism: apply the recorded bijection
  $\widehat{\sigma}_1 \circ \cdots \circ \widehat{\sigma}_k$ semantically and
  test the ANF endpoint --- exact, and indifferent to how the derivation was
  found.
\end{frame}

\begin{frame}{Backup: positioning}
  \begin{itemize}
    \item \textbf{maskVerif} \parencite{10.1007/978-3-030-29959-0_15}:
      language-based, the traversal's
      main inspiration; replay-style extension; no factoring.
    \item \textbf{REBECCA / Fourier-analytic} \parencite{bloem2018formal}:
      correlation sets over Fourier coefficients; approximate in general.
    \item \textbf{SILVER} \parencite{knichel2020silver}: exact, BDD-based model counting;
      exhaustive per-probe distributions, different scaling profile.
    \item \textbf{\toolname}: exact like SILVER, constructive like maskVerif ---
      and the certificate (the coupling) is itself an object with structure
      we exploit.
  \end{itemize}
\end{frame}

\begin{frame}{Backup: the four-share TI netlist}
  Four-share TI of $x \oplus (y \odot z)$ \parencite{bilgin2015threshold}; each input
  four-shared, sixteen cross products $y_i \odot z_j$:
  \[
  \begin{aligned}
    F_1 &= x_2 \oplus \textstyle\bigoplus_{i,j \in \{2,3,4\}} y_i z_j, &
    F_2 &= x_3 \oplus y_1 z_3 \oplus y_1 z_4 \oplus y_3 z_1 \oplus y_4 z_1 \oplus y_1 z_1, \\

  \]

\end{frame}
